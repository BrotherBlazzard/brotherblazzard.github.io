         var count = 0;

function multiply (  list  ,   multiplier  ){ 
   var     newList   = [] 
     for   (  var     i   =   0  ;    i   <   list  .  length  ;  ++  i  ) { 
       newList  [  i  ] =   list  [  i  ] *   multiplier     
     } 
    return     newList 
 }
